Electromagnetics

$$
d\vec B=\frac{\mu_0}{4\pi}\frac{Id\vec l\times\vec r}{r^3}\\
\vec {P_m}=IS\vec n\\
\oiint_S\vec B\cdot d\vec S=0\\
\oint_L\vec B\cdot d\vec l=\mu_0\sum I_i\\
$$

$$
直电流：\\
B=\frac{\mu_0I}{4\pi r}(\cos\theta_1-\cos\theta_2)\\
圆环电流：\
B=\frac{\mu_0IR^2}{2(x^2+R^2)^\frac32}\\
\vec {p_m}=IS\vec n\\
长直螺线管：\\
B=\mu_0nI\\
管外B\approx0\\
无限大载流平面：\\
B=\frac12\mu_0i\\
$$

$$
\vec F_e=q\vec E\\
\vec F_m=q\vec v\times\vec B\\
R=\frac{mv_\perp}{qB}\\
T=\frac{2\pi m}{qB}\\
d=v_{//}T\\
$$

$$
霍尔效应：\
I=qnvab\\
v=\frac{I}{qnab}\\
E_H=vB\\
V_H=\frac1{nq}\frac{IB}b=R_H\frac{IB}b\\
$$

$$
d\vec F=Id\vec l\times\vec B\\
线圈：\
\vec M=\vec {P_m}\times\vec B\\
\vec F=0\\
$$

$$
\vec B=\vec B_0+\vec {B’}\\
\mu_r=\frac{B}{B_0}\\
\vec{m_轨}=IS\vec n=-\frac{er^2}2\vec\omega\\
\vec{m_{分子}}=\sum\vec m_{轨}+\sum\vec m_自\
\begin{cases}
\vec m_{分子}\ne0\to顺磁质\mu_r\ge1\\
\vec m_{分子}=0\to抗磁质\mu_r<1\\
\end{cases}\\
\vec M=\frac{\sum\vec {m_i}}{\Delta V}\\
\oint_L\vec M\cdot d\vec l=\sum I’\\
\oint_L\vec B\cdot d\vec l=\mu_0\sum I+\mu_0\sum I’\\
\vec H=\frac{\vec B}{\mu_0}-\vec M\\
\oint_L \vec H\cdot d\vec l=\sum I_i\\
H(A/m)\\
\vec B=\mu_0\mu_r\vec H=\mu \vec H\\
\vec M=\chi_m\vec H=(\mu_r-1)\vec H\\
\vec {i’}=\vec M\times\vec{e_n}\\
$$

$$
饱和磁感应强度B_s\\
剩磁B_r\\
矫顽力H_c\\
$$

$$
\varepsilon_i=-\frac{d\psi}{dt}\\
1V=1Wb/s\\
\Phi_B=\int_S\vec B\cdot d\vec S\\
\varepsilon=A=\int_-^+\vec E_K\cdot d\vec l\\
动生：\
\vec E_k=\frac{f_洛}{-e}=\vec v\times \vec B\\
感生：\
\oint\vec E_i\cdot d\vec l=-\iint_S\frac{\partial\vec B}{\partial t}\cdot d\vec S\\
\varepsilon=-\int_S\frac{\partial\vec B}{\partial t}\cdot d\vec S+\int_L(\vec v\times\vec B)\cdot d\vec l=-\frac{d\Phi}{dt}\\
$$

$$
\psi_2=M_{21}i_1\\
M_{12}=M_{21}=M\\
\varepsilon_M=-\frac{d\psi}{dt}=-M\frac{di}{dt}-i\frac{dM}{dt}\\
\varepsilon_2=-M\frac{di_1}{dt}\\
\varepsilon_L=-\frac{d\psi}{dt}\\
\psi=LI\\
L(亨利H)\\
$$

$$
W_m=\frac12LI^2\\
w_m=\frac{W_m}V=\frac12\vec B\cdot\vec H\\
$$

$$
c=\frac1{\sqrt{\varepsilon_0\mu_0}}\\
\vec D=\varepsilon\vec E\\
\vec H=\frac{\vec B}{\mu}\\
I_D=\frac{d\Phi_D}{dt}=\int\frac{\partial\vec D}{\partial t}\cdot d\vec S\\
\vec j_D=\frac{\partial\vec D}{\partial t}=\varepsilon\frac{\partial\vec E}{\partial t}\\
\begin{cases}
\oint\vec D\cdot d\vec S=\int_V\rho dV\\
\oint\vec E\cdot d\vec l=-\int_S\frac{\partial\vec B}{\partial t}\cdot d\vec S\\
\oint\vec B\cdot d\vec S=0\\
\oint\vec H\cdot d\vec l=\int_S(\vec {j_{传导}}+\frac{\partial\vec D}{\partial t})\cdot d\vec S\\
\end{cases}\\
$$

$$
w=w_e+w_m=\frac12\vec D\cdot\vec E+\frac12\vec B\cdot\vec H\\
\vec S=\vec E\cdot\vec H\\
